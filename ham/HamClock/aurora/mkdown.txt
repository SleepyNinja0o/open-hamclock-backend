#!/bin/bash
# create dummy files containing a message

L1="Aurora is down"
L2='www.swpc.noaa.gov/products/aurora-30-minute-forecast'

mkimg () {

    FN=$1
    W=$2
    H=$3

    let "L1X=$W/4"
    let "L1Y=$H/3"
    let "L1S=$H/12"

    let "L2X=$W/18"
    let "L2Y=5*$H/9"
    let "L2S=$H/24"

    pbmmake -black $W $H \
        | ppmlabel -color "#ff0000" \
                -x $L1X -y $L1Y -size $L1S -text "$L1" \
                -x $L2X -y $L2Y -size $L2S -text "$L2" \
        | ../fetchVOACAPArea.d/p6tobmp565 > $FN

}

mkimg ../maps/map-N-7920x3960-Aurora.bmp 7920 3960
