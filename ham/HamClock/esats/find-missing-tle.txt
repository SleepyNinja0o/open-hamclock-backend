#!/bin/bash
# compare esats.subset and x.tmp

tmp=x.tmp.tmp

awk 'length($0){print}' x.tmp | sort > $tmp

EX=0
while read sline; do
    if [[ $sline != \#* ]] ; then
        if ! grep -q "$sline" $tmp; then
            echo no $sline in x.tmp
            EX=1
        fi
    fi
done < esats.subset

rm $tmp

exit $EX
